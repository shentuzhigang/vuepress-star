yarn add - D @starzkg/vuepress-theme-star

# 新建一个 markdown 文件
echo '# Hello VuePress Theme Star' > README.md

# 开始写作
yarn vuepress dev

# 构建静态文件
yarn vuepress build

npx vuepress dev
